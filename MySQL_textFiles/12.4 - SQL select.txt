SELECT ISS.NAME, I.SUMMARY, I.DESCRIPTION
FROM issueslists ISS
JOIN issues I ON I.ISSUESLIST_ID = ISS.ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM issues I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME;

/*for fun :)*/
SELECT ISS.NAME, I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM issues I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
              JOIN issueslists ISS ON I.ISSUESLIST_ID = ISS.ID
ORDER BY U.ID, ISS.ID;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUE_NUMBER
FROM issues i
join USERS U on i.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID
HAVING COUNT(*) > 0
ORDER BY U.ID;
